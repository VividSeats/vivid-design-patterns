# Backdrop

## Installation

Using the react components:

    import Backdrop from '@vividseats/vivid-design-patterns/dist/atoms/Backdrop';

Using the scss files:

    @import '@vividseats/vivid-design-patterns/src/styles/_backdrop.scss';

## Props

<Props of={Backdrop} />

## Usage

May be used on its own.

<Playground>
    <StatefulComponent initialState={{open:true}}>
            {
        ({state, setState}) => {
            const { open } = state;

            return (
                <React.Fragment>
                    <Button style={{zIndex: 10000000}} onClick={() => setState({open: !open})}>{open ? 'Close' : 'Open'}</Button>
                    <div style={{position: 'relative', height: '400px'}}>
                        <Backdrop isOpen={open} />
                    </div>
                </React.Fragment>
            )
        }
    }
        </StatefulComponent>
