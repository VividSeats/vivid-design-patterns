## Installation

Using the react components:

    import Carousel from '@vividseats/vivid-design-patterns/dist/molecule/Carousel';

or

    import { Carousel } from '@vividseats/vivid-design-patterns/dist/'

styles:

    @import '@vividseats/vivid-design-patterns/src/styles/carousel.scss';
