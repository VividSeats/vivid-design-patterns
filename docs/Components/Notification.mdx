# Notification

## Installation

Using the react components:

    import Notification from '@vividseats/vivid-design-patterns/dist/molecules/Notification';

Using the scss files:

    @import "@vividseats/vivid-design-patterns/src/styles/notification.scss";

## Props

<Props of={Notification} />

## Basic Usage

<Playground>
    <StatefulComponent initialState={{open: false}}>
            {
        ({state, setState}) => {
            const { open } = state;

            const handleClose = () => {
                setState({open: false})
            }

            const toggleNotification = () => {
                setState({open: !open})
            }

            return (
                <Fragment>
                    <Button onClick={toggleNotification}>Toggle Notification</Button>
                    <Notification hasBackdrop isOpen={open} onClickClose={handleClose}>
                        <Headline state="inverted" importance={3}>Vivid Seats </Headline>
                        <Headline state="inverted" importance={3}>Buy In-App,</Headline>
                        <Headline state="inverted" importance={3}>Earn Credit Back</Headline>
                    </Notification>
                </Fragment>
            )
        }
    }
    </StatefulComponent>

</Playground>

## Without an 'X' button

If no `onClickClose` callback is passed, the notification component will not render a 'X' button

<Playground>
    <StatefulComponent initialState={{open: false}}>
        {
                ({state, setState}) => {
                    const { open } = state;
                    const handleClose = () => {
                        setState({open: false})
                    }

                    const toggleNotification = () => {
                        setState({open: !open})
                    }

                    return (
                        <Fragment>
                            <Button onClick={toggleNotification}>Toggle Notification</Button>
                            <Notification isOpen={open}>
                                <Headline state="inverted" importance={3}>Vivid Seats </Headline>
                                <Headline state="inverted" importance={3}>Buy In-App,</Headline>
                                <Headline state="inverted" importance={3}>Earn Credit Back</Headline>
                                <Button onClick={handleClose}> Close </Button>
                    </Notification>
                </Fragment>
            )
        }
    }
    </StatefulComponent>

</Playground>
