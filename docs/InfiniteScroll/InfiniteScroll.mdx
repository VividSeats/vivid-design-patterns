## Basic Usage

<Playground>
    {
        () => {
            let events = [];
            for(let i = 0; i < 300; i++ ) {
                events.push(getRandomEvent())
            }

            return (
                <InfiniteScroll>
                    {
                        events.map((event) => {
                            return <EventRow subtitle={event.eventLocation} href='!#' title={event.eventName} key={`${event.eventName}-${event.eventLocation}-${event.id}`}/>
                        })
                    }
                </InfiniteScroll>
            )
        }
    }
</Playground>
